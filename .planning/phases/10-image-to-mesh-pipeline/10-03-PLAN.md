   export async function submitReconstruction(
     images: File[],
     processTypes?: string,
     rulePack?: string
   ): Promise<{ job_id: string; status: string; poll_url: string; estimated_seconds: number }> {
     const form = new FormData()
     images.forEach(img => form.append('images', img))
     if (processTypes) form.append('process_types', processTypes)
     if (rulePack) form.append('rule_pack', rulePack)
     const res = await apiClient.post('/api/v1/reconstruct', form)
     return res.data
   }
   ```

   }
   ```
